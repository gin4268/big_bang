配置：
解释说明：
在config文件中配置

初始化地图：
var aMap = api.require('aMap');
aMap.open({
    rect: {
        x: 0,
        y: 0,
        w: 320,
        h: 300
    },
    showUserLocation: true,
    zoomLevel: 11,
    center: {
        lon: 116.4021310000,
        lat: 39.9994480000
    },
    fixedOn: api.frameName,
    fixed: true
}, function(ret, err) {
    if (ret.status) {
        alert(JSON.stringify(ret));
    } else {
        alert(JSON.stringify(err));
    }
});

设置地图属性：
aMap.setMapAttr({
    type: 'standard',  // 标准,satellite,卫星,night,夜间
    trafficOn: true,
    zoomEnable: false,
    scrollEnable: false,
    building: true,
    overlookEnabled: false,
    rotateEnabled: false
});

添加事件：
var aMap = api.require('aMap');
aMap.addEventListener({
    name: 'longPress'
}, function(ret) {
    if (ret.status) {
        alert(JSON.stringify(ret));
    }
});
解释说明：
// longPress(长按事件),viewChange(位置改变事件),click(单击事件),zoom(缩放事件)

添加标注：
var aMap = api.require('aMap');
aMap.addAnnotations({
    annotations: [{
        id: 1,
        lon: 116.297,
        lat: 40.109
    }],
    icons: ['widget://'],
    draggable: true,
    timeInterval: 2.0
}, function(ret) {
    if (ret.eventType == 'click') {
        alert(ret.id);
    }
});
