偏好值(全局数据)：
// 异步返回结果 
api.getPrefs({
	key: 'key'
}, function(ret, err) {
	var username = ret.value;
});
// 同步返回结果 
var username = api.getPrefs({
	sync: true,
	key: 'key'
});
// 设置偏好值 
api.setPrefs({
    key: 'k',
    value: '1'
});

全局事件：
// 发送一个全局事件 
api.sendEvent({
	name: 'keyWord',
	extra: {  // 发送参数
		key1: inval,
		key2: inval2,
		key3: inval3,
		key4: inval4,
		key5: inval5,
	}
});
// 获取全局事件 
api.addEventListener({
	name: 'keyWord'
}, function(ret, err){
	value = ret.value;  // 获取参数
	inval = value.key1;
});
解释说明：

页面操作：
// 打开一个页面,传入参数 
api.openWin({
   name:'wzsjlr0',
   url:'wzsjlr0.html',
   animation:{
	  type:"push",                //动画类型（详见动画类型常量）
	  subType:"from_bottom",       //动画子类型（详见动画子类型常量）
	  duration:300
   },
   pageParam : {  // 打开页面的时候传入参数,可传json类型
		key : val
	bgColor:'#FFF'
});
// 获取页面参数 
api.pageParam.info
// 关闭页面 
api.closeWin({
	name: 'page1'
});
// 关闭所有页面 
api.closeToWin({
	name: 'root',
	animation:{
		type:"fade",                //动画类型（详见动画类型常量）
		subType:"from_right",       //动画子类型（详见动画子类型常量）
		duration:300                //动画过渡时间，默认300毫秒
	}
});
解释说明：
