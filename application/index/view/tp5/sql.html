根据某字段的值查询 模糊查询 or条件查询 返回某个字段的值 数据集分批处理 where条件 多条件查询 获取记录条数 in条件查询
根据某字段的值查询：
// 根据某字段的值查询
->where('status',1)
解释说明：
多个->where()可表示and

模糊查询：
// 模糊查询 
->where('name','like','%thinkphp')
// 多字段模糊查询 
->where('name&title','like','%thinkphp')
// 分离的模糊查询 
$where['username'] = array('like', "%$cname%");
解释说明：

or条件查询：
// or条件查询 
->where()->whereOr()
// 多字段or条件模糊查询 
->where('name|title','like','%thinkphp')
解释说明：
->where()->where()表示and条件查询

返回某个字段的值：
// 查询某一列的值的数组 
Db::table('think_user')->where('status',1)->column('name');
// 返回某一列的第一个值 
Db::table('think_user')->value('name');
解释说明：

数据集分批处理：
Db::table('think_user')->where('score','>',80)->chunk(100, function($users) {
    foreach ($users as $user) {
    	return false;  // 终止处理
    }
});

where条件：
// 等于 
where('id','eq',100)
// 不等于 
where('id','neq',100)
// 大于 
where('id','gt',100)
// 小于 
where('id','lt',100)

where('id','between','1,8')

where('id','not in','1,5,8')

where('title','null')

where('name','not null')
